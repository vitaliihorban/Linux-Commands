Версія Git: # git --version

Конфігурація(глобальна):
- usename: # git config --global user.name "Volodymyr Zavada"
- email:  # git config --global user.email "volodymyrzavada@gmail.com"

Ініціалізація репозиторію: # git init

Створити файл з консолі: # cat >> test.txt(enter) і вписуємо текст(напр: Hello Git). Для виходу - Ctrl+D

Перевірка стану: # git status
Додавання файлу до Git(комміт кандидат): # git add назва файлу або паттерн (дадати все . або *)
Створення комміту: # git commit -m "First Commit"

Перегляд історії коммітів: # git log
- опції:
# git log --pretty=oneline (в 1 рядок інфо про кожен комміт) 
# git log --max-count=3 (останні 3 комміти)
# git log --all (виводить всі комміти які було зроблено)
# git log --author="Volodymyr Zavada" (вивід всіх коммітів за user.name автора)
# git log --pretty=format:"%h - %s : %ad [ %an ]" --date=short (задати власний формат виводу https://mirrors.edge.kernel.org/pub/software/scm/git/docs/git-show.html#_pretty_formats)

Переключення між коммітами: # git checkout хеш-комміту

Відміна індексації файла/ів(відміна комміт кандидата, видалення проіндексованого файла): 
# git reset  HEAD test.txt (HEAD - останній стан, замість HEAD - можна хеш комміту)
# git checkout назва файлу/ів (відмінити всі зміни(скинути стан) до останнього комміту)

Відміна комміту(повернення стану до вказаного комміту): 
# git revert HEAD --no-edit (Замість HEAD - можна хеш комміту до якого відкатуємо зміни - може виникнути конфлікт)

Гілки:
# git branch (показує всі гілки і на якій ми знаходимось)
# git checkout -b назва_гілки (створення нової гілки)
# git checkout назва_гілки (перехід між гілками)

Злиття гілок(потрібно перейти на ту гілку в яку будуть перекинуті зміни):
# git merge назва_гілки_з_якої_беремо_зміни

Отримання змін:
# git pull [--rebase] origin master // rebase - будує історію за часом створення комітів

# Створюємо локальний репозиторій і добавляємо readme.md 
$ git init

$ cat >> README.md (нажати Enter і ввести текст. Для збереження Ctrl+D)  
---- Приклад: 
---- cat >> README.md
---- some text
---- Ctrl+D

# Як і вище. В цьому файлі вказуємо всі папки і файли які Git буде ігнорувати і не закине їх на Github
$ cat >> .gitignore 

# крапка обовязкова - добавить всі зміни в репозиторії, окім тих що в gitignore файлі
$ git add . 

# Зєднуємо локальний репозиторій з репозиторієм на GitHub(потрібен URL)
# На Github має бути створений пустий репозиторій
$ git remote add origin https://github.com/c0ldlimit/vimcolors.git

# Коммітим зміни на git. Перший(базовий) комміт принято називати "Initial commit"
# якщо 1й раз через GitBush - зявиться вікно де потрібно ввести емейл/пароль
$ git commit -m "Initial commit"

# закине ваші зміни на Github
$ git push -u origin master

# Тепер можемо робити зміни в локальному репозиторії і просто закидувати їх на Github
# kрапка обовязкова - добавить всі зміни в репозиторії, окім тих що в gitignore файлі
# окрім крапки можна писати назви файлів або паттерни. Прочитайте про це
# наступні 4 команди найчастіше використовуватимуться
$ git add .  

# також, звикайте перевіряти, які файли були додані
$ git status

# -am не розділяйте пробілом. Повідомлення в ".." не може повторюватись 
$ git commit -am "Some message"

# закине ваші зміни на Github 
$ git push origin master

=================================================================
# Якщо вам потрібно скопіювати репозиторій з Github
# після виконання команди нижче, буде створена папка 
# з вмістом вашого репозиторію на Github (URL міняєте на свій)
$ git clone https://github.com/c0ldlimit/vimcolors.git
# Якщо виконаєте якісь зміни - використовуйте 3 команди як описано вище
=================================================================

# Для отримання змін з віддаленого репозиторію(GitHub):
$ git pull origin master

# крапка вказує на всі файли(будуть видалені з кешу, але в репозиторії залишаться)
$ git rm -r --cached .
